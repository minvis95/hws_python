# 3. 계단 만들기

number = int(input())

# 아래에 코드를 작성하시오.
cnt1, cnt2 = 1, 0
number_list = list(range(1, number + 1))

while cnt1 <= number:
    for i in number_list:
    	if cnt2 < cnt1:
        	print('{}'.format(i), end=' ')
            cnt2 += 1
            continue
        else:
            break
    print('\0')
    cnt2 = 0
    cnt1 += 1
