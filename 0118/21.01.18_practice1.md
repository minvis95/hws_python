```python
# 최댓값과 등장 횟수 구하기

numbers = [7, 10, 22, 7, 22, 22]

# 아래에 코드를 작성하시오.
cnt, max_number = 1, 0

for i in numbers:
    if i > max_number:
        max_number = i
        cnt = 1
        continue
    if i == max_number:
        cnt += 1
print(max_number, cnt)
```

```python
# 'a' 가 싫어

word = input()

# 아래에 코드를 작성하시오.

# cnt = len(word)
cnt = 0
for i in word:
    cnt += 1
    
result = str()
for i in range(cnt):
    if word[i] == 'a':
        continue
    result += word[i]
print(result)
```

```python
# 단어 뒤집기

# 아래에 코드를 작성하시오.
# print(word[::-1])

# cnt = len(word)
cnt = 0
for i in word:

    result += word[i]
```
